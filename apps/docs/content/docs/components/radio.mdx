# Radio

Radio component for single-choice selection controls. Perfect for forms where users need to select one option from a set of mutually exclusive choices.

## Installation

```bash
npm install forteui-core
```

## Import

```tsx
import { Radio, RadioGroup } from 'forteui-core';
```

## Basic Usage

```tsx
import { Radio, RadioGroup } from 'forteui-core';

function MyComponent() {
  return (
    <RadioGroup defaultValue="option1">
      <Radio value="option1" label="Option 1" />
      <Radio value="option2" label="Option 2" />
      <Radio value="option3" label="Option 3" />
    </RadioGroup>
  );
}
```

## Controlled RadioGroup

```tsx
import { useState } from 'react';

function ControlledRadioExample() {
  const [value, setValue] = useState('option2');
  
  return (
    <RadioGroup value={value} onValueChange={setValue}>
      <Radio value="option1" label="Option 1" />
      <Radio value="option2" label="Option 2" />
      <Radio value="option3" label="Option 3" />
    </RadioGroup>
  );
}
```

## Form Integration

### Survey Form Example

```tsx
import { useState } from 'react';

function SurveyForm() {
  const [satisfaction, setSatisfaction] = useState('');
  const [frequency, setFrequency] = useState('');
  
  return (
    <form className="space-y-6">
      <div>
        <h3 className="text-lg font-semibold mb-3">How satisfied are you?</h3>
        <RadioGroup value={satisfaction} onValueChange={setSatisfaction}>
          <Radio value="very-satisfied" label="Very Satisfied" variant="success" />
          <Radio value="satisfied" label="Satisfied" variant="primary" />
          <Radio value="neutral" label="Neutral" variant="default" />
          <Radio value="dissatisfied" label="Dissatisfied" variant="warning" />
          <Radio value="very-dissatisfied" label="Very Dissatisfied" variant="error" />
        </RadioGroup>
      </div>
      
      <div>
        <h3 className="text-lg font-semibold mb-3">How often do you use our product?</h3>
        <RadioGroup value={frequency} onValueChange={setFrequency}>
          <Radio value="daily" label="Daily" />
          <Radio value="weekly" label="Weekly" />
          <Radio value="monthly" label="Monthly" />
          <Radio value="rarely" label="Rarely" />
        </RadioGroup>
      </div>
    </form>
  );
}
```
