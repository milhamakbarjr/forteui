# Slider

A versatile range input component for selecting numeric values within a specified range. Supports single and dual handle modes, step controls, custom formatting, and accessibility features.

## Installation

```bash
npm install forteui-core
```

## Import

```tsx
import { Slider } from 'forteui-core';
```

## Basic Usage

```tsx
import { Slider } from 'forteui-core';

function MyComponent() {
  return (
    <Slider 
      min={0}
      max={100}
      defaultValue={50}
      label="Volume"
    />
  );
}
```

## Single Handle Slider

### Basic Single Slider

```tsx
import { useState } from 'react';

function SingleSliderExample() {
  const [value, setValue] = useState(25);
  
  return (
    <div className="space-y-6">
      <Slider 
        label="Temperature"
        min={0}
        max={100}
        value={value}
        onChange={setValue}
        suffix="°C"
      />
      
      <p className="text-sm text-gray-600">
        Current temperature: {value}°C
      </p>
    </div>
  );
}
```

### With Step Control

```tsx
<div className="space-y-4">
  <Slider

  <Slider 
    label="Rating (step: 0.5)"

    max={5}
    step={0.5}
    defaultValue={3.5}

```

## Controlled vs Uncontrolled

### Controlled Slider

```tsx
import { useState } from 'react';

function ControlledSlider() {
  const [value, setValue] = useState(50);
  
  return (
    <Slider 
      label="Controlled"
      value={value}
      onChange={setValue}
      showValue
    />
  );
}
```

### Uncontrolled Slider

```tsx
function UncontrolledSlider() {
  return (
    <Slider 
      label="Uncontrolled"
      defaultValue={75}
      showValue
    />
  );
}
```

## Event Handling

### onChange and onChangeEnd

```tsx
import { useState } from 'react';

function EventHandlingExample() {
  const [value, setValue] = useState(50);
  const [endValue, setEndValue] = useState(50);
  
  return (
    <div className="space-y-4">
      <Slider 
        label="Volume"
        value={value}
        onChange={setValue}
        onChangeEnd={setEndValue}
        showValue
      />
      
      <div className="text-sm text-gray-600">
        <p>Current value: {value}</p>
        <p>End value: {endValue}</p>
      </div>
    </div>
  );
}
```
