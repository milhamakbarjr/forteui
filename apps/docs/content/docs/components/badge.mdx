# Badge

Badge wrapper component for displaying count numbers or dot indicators on top of child elements. Perfect for notification badges on icons, unread message counts, and status indicators.

## Installation

```bash
npm install forteui-core
```

## Import

```tsx
import { Badge } from 'forteui-core';
```

## Basic Usage

```tsx
import { Badge } from 'forteui-core';
import { IconMail } from 'forteui-core';

function MyComponent() {
  return (
    <Badge variant="primary" count={5}>
      <IconMail size={24} />
    </Badge>
  );
}
```

## Interactive Examples

### Real-time Counter

```tsx
import { useState } from 'react';
import { Badge } from 'forteui-core';
import { IconMail, IconShoppingCart } from 'forteui-core';

function InteractiveExample() {
  const [messageCount, setMessageCount] = useState(3);
  const [cartCount, setCartCount] = useState(0);
  
  return (
    <div className="flex gap-6">
      <div className="flex flex-col items-center gap-2">
        <Badge variant="primary" count={messageCount}>
          <IconMail size={32} />
        </Badge>
        <button 
          onClick={() => setMessageCount(prev => prev + 1)}
          className="px-3 py-1 bg-blue-100 rounded text-sm"
        >
          Add Message ({messageCount})
        </button>
      </div>
      
      <div className="flex flex-col items-center gap-2">
        <Badge variant="success" count={cartCount} showZero>
          <IconShoppingCart size={32} />
        </Badge>
        <button 
          onClick={() => setCartCount(prev => prev + 1)}
          className="px-3 py-1 bg-green-100 rounded text-sm"
        >
          Add to Cart ({cartCount})
        </button>
      </div>
    </div>
  );
}
```

## Advanced Features

### showZero Prop

```tsx
import { Badge } from 'forteui-core';
import { IconMail } from 'forteui-core';

function ShowZeroExample() {
  return (
    <div className="flex gap-4">
      <Badge variant="primary" count={0}>
        <IconMail size={24} />
      </Badge>
      
      <Badge variant="primary" count={0} showZero>
        <IconMail size={24} />
      </Badge>
    </div>
  );
}
```

### Custom max Value

```tsx

function CustomMaxExample() {

      <Badge variant="warning" count={150} max={50}>

      </Badge>
      
      <Badge variant="error" count={1500} max={999}>
        <IconAlert size={24} />
      </Badge>
    </div>

}
```

## Real-World Examples

### Notification System

```tsx
import { useState } from 'react';
import { Badge, Button } from 'forteui-core';
import { IconBell, IconMail, IconShoppingCart } from 'forteui-core';

function NotificationSystem() {
  const [notifications, setNotifications] = useState(3);
  const [messages, setMessages] = useState(12);
  const [cartItems, setCartItems] = useState(0);
  
  return (
    <div className="flex gap-4">
      <Badge variant="error" count={notifications}>
        <Button variant="outline-primary">
          <IconBell size={20} />
        </Button>
      </Badge>
      
      <Badge variant="primary" count={messages}>
        <Button variant="outline-primary">
          <IconMail size={20} />
        </Button>
      </Badge>
      
      <Badge variant="success" count={cartItems} showZero>
        <Button variant="outline-primary">
          <IconShoppingCart size={20} />
        </Button>
      </Badge>
    </div>
  );
}
```
