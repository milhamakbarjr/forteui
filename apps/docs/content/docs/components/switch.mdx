# Switch

Switch component for binary on/off controls. Perfect for settings, toggles, and feature enablement with smooth animations and excellent user experience.

## Installation

```bash
npm install forteui-core
```

## Import

```tsx
import { Switch } from 'forteui-core';
```

## Basic Usage

```tsx
import { Switch } from 'forteui-core';

function MyComponent() {
  return (
    <Switch 
      label="Enable notifications"
      defaultChecked={false}
    />
  );
}
```

## Controlled Switch

```tsx
import { useState } from 'react';

function ControlledSwitchExample() {
  const [isEnabled, setIsEnabled] = useState(false);
  
  return (
    <div className="space-y-4">
      <Switch 
        label="Dark Mode"
        checked={isEnabled}
        onCheckedChange={setIsEnabled}
      />
      <p className="text-sm text-gray-600">
        Dark mode is {isEnabled ? 'enabled' : 'disabled'}
      </p>
    </div>
  );
}
```

## Settings Panel Example

```tsx
import { useState } from 'react';

function SettingsPanel() {
  const [settings, setSettings] = useState({
    notifications: true,
    emailUpdates: false,
    autoSave: true,
    darkMode: false,
    analytics: true,
  });
  
  const updateSetting = (key: string) => (checked: boolean) => {
    setSettings(prev => ({ ...prev, [key]: checked }));
  };
  
  return (
    <div className="space-y-6 p-6 bg-white rounded-lg border">
      <h3 className="text-lg font-semibold">Preferences</h3>
      
      <div className="space-y-4">
        <Switch 
          label="Push Notifications"
          checked={settings.notifications}
          onCheckedChange={updateSetting('notifications')}
          color="primary"
        />
        
        <Switch 
          label="Email Updates"
          checked={settings.emailUpdates}
          onCheckedChange={updateSetting('emailUpdates')}
          color="info"
        />
        
        <Switch 
          label="Auto Save"
          checked={settings.autoSave}
          onCheckedChange={updateSetting('autoSave')}
          color="success"
        />
        
        <Switch 
          label="Dark Mode"
          checked={settings.darkMode}
          onCheckedChange={updateSetting('darkMode')}
          color="default"
        />
        
        <Switch 
          label="Analytics"
          checked={settings.analytics}
          onCheckedChange={updateSetting('analytics')}
          color="warning"
        />
      </div>
    </div>
  );
}
```

## Form Integration

```tsx
import { useState } from 'react';

function ProfileForm() {
  const [profile, setProfile] = useState({
    isPublic: false,
    allowMessages: true,
    showActivity: false,
  });
  
  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    console.log('Profile settings:', profile);
  };
  
  return (
    <form onSubmit={handleSubmit} className="space-y-6">
      <div className="space-y-4">
        <Switch 
          label="Public Profile"
          checked={profile.isPublic}
          onCheckedChange={(checked) => 
            setProfile(prev => ({ ...prev, isPublic: checked }))
          }
        />
        
        <Switch 
          label="Allow Direct Messages"
          checked={profile.allowMessages}
          onCheckedChange={(checked) => 
            setProfile(prev => ({ ...prev, allowMessages: checked }))
          }
        />
        
        <Switch 
          label="Show Activity Status"
          checked={profile.showActivity}
          onCheckedChange={(checked) => 
            setProfile(prev => ({ ...prev, showActivity: checked }))
          }
        />
      </div>
      
      <Button type="submit" color="primary">
        Save Settings
      </Button>
    </form>
  );
}
```
